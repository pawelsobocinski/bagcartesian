\begin{abstract}
We introduce the notion of \emph{bag-cartesian bicategory} --- the mono-stable fragment of the cartesian bicategory axioms of Carboni--Walters --- and prove a characterisation theorem: a functionally complete bag-cartesian bicategory is biequivalent to $\Span_{\mathrm{mono}}(\Map(\mathbf{B}))$, the bicategory of spans with monomorphic 2-cells. As a corollary, the bag-cartesian axioms are \emph{complete} for $\Mat_\N = \Span_{\mathrm{mono}}(\FinSet)$, resolving the Chaudhuri--Vardi problem: bag-semantic conjunctive query containment is decidable.
\end{abstract}

\section{Introduction}
%% ============================================================

The containment problem for conjunctive queries (CQs) under bag semantics has been open since Chaudhuri and Vardi raised it in 1993 \cite{CV}. Under set semantics, the problem is decidable and elegantly characterised: containment coincides with the existence of a homomorphism between query hypergraphs (Chandra--Merlin \cite{CM}), which in turn coincides with derivability in the equational theory of cartesian bicategories (Bonchi--Seeber--Soboci\'nski \cite{BSS}). Under bag semantics, no such characterisation has been known.

We develop a framework that extends the cartesian bicategory approach to bag semantics. The central observation is that $\Mat_\N$, the bicategory of $\N$-valued matrices modelling bag semantics, arises as $\Span_{\mathrm{mono}}(\FinSet)$: the locally posetal bicategory obtained from $\Span(\FinSet)$ by restricting 2-cells to monomorphisms. The axioms of what we call a \emph{bag-cartesian bicategory} are precisely the \emph{mono-stable fragment} of the cartesian bicategory axioms of Carboni--Walters \cite{CW,CKWW}: those axioms whose witnessing 2-cells in $\Span$ are necessarily injective.

\subsection{Conventions}

Throughout, $\mathbf{B}$ denotes a locally posetal bicategory with symmetric monoidal structure $(\otimes, I)$. We write composition in \textbf{diagrammatic order}: $f;g$ means ``first $f$, then $g$.'' We write $f \colon A \to B$ for an arrow with source $A$ and target $B$.

An arrow $f \colon A \to B$ is a \emph{map} if it has a right adjoint $f^* \colon B \to A$ satisfying
\[
  \text{unit: } \id_A \leq f;f^*, \qquad
  \text{counit: } f^*;f \leq \id_B.
\]
We write $\Map(\mathbf{B})$ for the locally full sub-bicategory of maps.

\section{Bag-Cartesian Bicategories}
\label{sec:def}
%% ============================================================

We build the definition in stages: first the Frobenius structure, then the ordering axioms.

\subsection{Special commutative Frobenius structure}

\begin{definition}[Frobenius structure]\label{def:frob}
A \emph{commutative special Frobenius structure} on an object $X$ of $\mathbf{B}$ consists of four arrows
\[
  \delta_X \colon X \to X \otimes X, \quad
  \mu_X \colon X \otimes X \to X, \quad
  \varepsilon_X \colon X \to I, \quad
  \eta_X \colon I \to X
\]
satisfying the following equations.

\begin{enumerate}[label=(\alph*),leftmargin=2em]
  \item \textbf{Comonoid.} $(\delta_X, \varepsilon_X)$ is a cocommutative comonoid:
  \begin{gather*}
    \delta;(\delta \otimes \id) = \delta;(\id \otimes \delta), \qquad
    \delta;\sigma = \delta, \\
    \delta;(\varepsilon \otimes \id) = \id = \delta;(\id \otimes \varepsilon),
  \end{gather*}
  where $\sigma \colon X \otimes X \to X \otimes X$ is the symmetry.

  \item \textbf{Monoid.} $(\mu_X, \eta_X)$ is a commutative monoid:
  \begin{gather*}
    (\mu \otimes \id);\mu = (\id \otimes \mu);\mu, \qquad
    \sigma;\mu = \mu, \\
    (\eta \otimes \id);\mu = \id = (\id \otimes \eta);\mu.
  \end{gather*}

  \item \textbf{Frobenius law.}
  \[
    (\delta \otimes \id);(\id \otimes \mu) = \mu;\delta = (\id \otimes \delta);(\mu \otimes \id).
  \]

  \item \textbf{Special law.} $\delta;\mu = \id_X$.
\end{enumerate}
\end{definition}

\begin{remark}\label{rmk:spider}
In a commutative special Frobenius algebra, every connected string diagram with $m$ inputs and $n$ outputs is equal to a unique \emph{spider} $\mathrm{Spider}(m,n)$. This is the ``spider theorem,'' which gives a normal form for connected diagrams. In particular, $\mu;\varepsilon = \mathrm{Spider}(2,0)$ and $\eta;\delta = \mathrm{Spider}(0,2)$.
\end{remark}

\subsection{The Frobenius transpose}

The Frobenius structure provides a canonical way to ``bend wires,'' yielding a transpose operation that will play a central role.

\begin{definition}[Frobenius transpose]\label{def:transpose}
For an arrow $r \colon X \to Y$, the \emph{Frobenius transpose} $r^\op \colon Y \to X$ is defined by
\[
  r^\op \;=\; ((\eta_X;\delta_X) \otimes \id_Y);(\id_X \otimes r \otimes \id_Y);(\id_X \otimes (\mu_Y;\varepsilon_Y))
\]
where $\eta_X;\delta_X \colon I \to X \otimes X$ creates a ``cup'' on $X$ and $\mu_Y;\varepsilon_Y \colon Y \otimes Y \to I$ creates a ``cap'' on $Y$. Equivalently (by the spider theorem), $r^\op$ is obtained by rotating the string diagram of $r$ by $180^\circ$.
\end{definition}

\begin{lemma}\label{lem:transpose-props}
The Frobenius transpose satisfies:
\begin{enumerate}[label=(\roman*)]
  \item $(r^\op)^\op = r$ \quad \textup{(involution)},
  \item $(r;s)^\op = s^\op; r^\op$ \quad \textup{(contravariance)},
  \item $\id^\op = \id$,
  \item for an injective map $h$: $h^\op = h^*$ \quad \textup{(transpose $=$ right adjoint)}.
\end{enumerate}
\end{lemma}

\begin{proof}
Properties (i)--(iii) follow from the Frobenius equations and the special law by standard string diagram calculations. For (iv): if $h$ is an injective map with $h^*;h \leq \id$ and $\id \leq h;h^*$, then $h^\op$ satisfies the same adjunction conditions (verified by diagrammatic computation), so $h^\op = h^*$ by uniqueness of right adjoints in a poset.
\end{proof}

\subsection{The Hadamard product}

\begin{definition}[Hadamard product]\label{def:hadamard}
For arrows $f, g \colon X \to Y$, the \emph{Hadamard product} is
\[
  f \Had g \;=\; \delta_X;(f \otimes g);\mu_Y \;\colon\; X \to Y.
\]
\end{definition}

In $\Mat_\N$, $(f \Had g)(x,y) = f(x,y) \cdot g(x,y)$ (entrywise multiplication).

\subsection{The axioms}

We can now state the full definition.

\begin{definition}[Bag-cartesian bicategory]\label{def:bag-cart}
A locally posetal bicategory $\mathbf{B}$ with symmetric monoidal structure $(\otimes, I)$ is \emph{bag-cartesian} if:
\begin{enumerate}[label=\textbf{(\roman*)},leftmargin=2.5em]
  \item \textbf{Frobenius structure.} Each object $X \in \ob(\mathbf{B})$ carries a commutative special Frobenius structure $(\delta_X, \mu_X, \varepsilon_X, \eta_X)$ as in Definition~\ref{def:frob}.

  \item \textbf{Lax naturality of $\delta$.} For every arrow $r \colon X \to Y$:
  \[
    r;\delta_Y \;\leq\; \delta_X;(r \otimes r).
  \]
  Equivalently, using the special law: $r \leq r \Had r$ \textup{(``bag-lax'')}.

  \item \textbf{Cauchy--Schwarz.} For all $f \colon X \to Y$ and $g \colon Y \to Z$:
  \[
    (f \Had f);(g \Had g) \;\leq\; (f;g) \Had (f;g).
  \]

  \item \textbf{Special law.} $\delta_X;\mu_X = \id_X$ for all $X$ \textup{(already part of the Frobenius structure, listed for emphasis)}.

  \item \textbf{Unit-of-adjunction.} For all $X$:
  \[
    \id_X \;\leq\; \varepsilon_X;\eta_X.
  \]
\end{enumerate}
\end{definition}

\begin{remark}
Axiom (ii) is equivalent to lax naturality of $\delta$ composed with the special law. In $\Mat_\N$, it states $n \leq n^2$ for all $n \in \N$ (which holds since $0 \leq 0$ and $n \leq n^2$ for $n \geq 1$). Axiom (iii) states $\sum_y a_y^2 b_y^2 \leq (\sum_y a_y b_y)^2$ for non-negative integers --- a consequence of the expansion $(\sum a_y b_y)^2 = \sum a_y^2 b_y^2 + \sum_{y \neq y'} a_y b_y a_{y'} b_{y'}$ with non-negative cross terms. Axiom (v) states $[a=b] \leq 1$ in $\N$, i.e., the identity matrix is entrywise dominated by the all-ones matrix.
\end{remark}

\begin{remark}[What is \emph{not} assumed]
In contrast with the cartesian bicategories of Carboni--Walters \cite{CW}, we do \textbf{not} assume:
\begin{itemize}
  \item The full adjunction $\varepsilon \dashv \eta$ (only the unit half, axiom (v)). The counit $\eta;\varepsilon \leq \id_I$ \emph{fails} in $\Mat_\N$: $(\eta;\varepsilon)(*,*) = |X| \not\leq 1$ for $|X| > 1$.
  \item Lax naturality of $\varepsilon$: $r;\varepsilon_Y \leq \varepsilon_X$. This \emph{fails} in $\Mat_\N$: it would require all row sums $\leq 1$.
  \item Discreteness: $\delta;\delta^* = (\delta^* \otimes \id);(\id \otimes \delta)$.
\end{itemize}
\end{remark}

\subsection{The motivating example}

\begin{example}[$\Mat_\N$]\label{ex:mat-n}
The bicategory $\Mat_\N$ has finite sets as objects, $\N$-valued matrices $M \colon X \times Y \to \N$ as arrows $X \to Y$, and composition given by matrix multiplication: $(M;N)(x,z) = \sum_y M(x,y) \cdot N(y,z)$. The monoidal product is $X \otimes Y = X \times Y$ with $(M \otimes N)((x,y),(x',y')) = M(x,x') \cdot N(y,y')$, and $I = \{*\}$. The ordering is entrywise: $M \leq M'$ iff $M(x,y) \leq M'(x,y)$ for all $x,y$.

The Frobenius structure on an object $X$ is given by the following matrices:
\begin{align*}
  \delta_X &\colon X \to X \times X, &
  \delta_X(a,(b,c)) &= [a{=}b{=}c], \\
  \mu_X &\colon X \times X \to X, &
  \mu_X((a,b),c) &= [a{=}b{=}c], \\
  \varepsilon_X &\colon X \to I, &
  \varepsilon_X(a,*) &= 1, \\
  \eta_X &\colon I \to X, &
  \eta_X(*,a) &= 1,
\end{align*}
where $[P]$ denotes the Iverson bracket ($1$ if $P$ is true, $0$ otherwise). One verifies $\delta;\mu = \id$ (special law) and that axioms (i)--(v) all hold. The Hadamard product is entrywise multiplication: $(M \Had N)(x,y) = M(x,y) \cdot N(x,y)$. The Frobenius transpose is the matrix transpose: $M^\op(y,x) = M(x,y)$.

The maps in $\Mat_\N$ are precisely the injective functions: a $\{0,1\}$-matrix $M_h$ corresponding to a function $h \colon A \to B$ (with $M_h(a,b) = [h(a){=}b]$) satisfies $M_h^*;M_h \leq \id_B$ iff $|h^{-1}(b)| \leq 1$ for all $b$, i.e., $h$ is injective. As we shall see (Section~\ref{sec:mono-stability}), $\Mat_\N \cong \Span_{\mathrm{mono}}(\FinSet)$.
\end{example}

\section{The Frobenius Isomorphism}
\label{sec:frob-iso}
%% ============================================================

\begin{theorem}[Frobenius Isomorphism]\label{thm:frob-iso}
In a bag-cartesian bicategory $\mathbf{B}$, the map
\[
  \Phi \colon \mathbf{B}(X,Y) \to \mathbf{B}(X \otimes Y, I), \qquad
  \Phi(s) = (s \otimes \id_Y);\mu_Y;\varepsilon_Y
\]
is an order-isomorphism, with inverse
\[
  \Psi \colon \mathbf{B}(X \otimes Y, I) \to \mathbf{B}(X,Y), \qquad
  \Psi(q) = (\id_X \otimes \eta_Y);(\id_X \otimes \delta_Y);(q \otimes \id_Y).
\]
\end{theorem}

\begin{proof}
We show $\Psi(\Phi(s)) = s$. Expanding:
\[
  \Psi(\Phi(s)) = (\id \otimes \eta);(\id \otimes \delta);(((s \otimes \id);\mu;\varepsilon) \otimes \id).
\]
Tracing wires: $\eta$ creates a $Y$-wire; $\delta$ copies it into $y_1, y_2$. The copy $y_1$ enters $(s \otimes \id);\mu;\varepsilon$: it is paired with $s$'s output via $\mu$, then absorbed by $\varepsilon$. The copy $y_2$ passes to the output.

By the comonoid unit law $\delta;(\varepsilon \otimes \id) = \id$, the monoid unit law $(\eta \otimes \id);\mu = \id$, and the special law $\delta;\mu = \id$, the net effect is: $y_2$ equals $s$'s output. So $\Psi(\Phi(s)) = s$. The identity $\Phi(\Psi(q)) = q$ is symmetric.
\end{proof}

\begin{remark}
This isomorphism replaces the compact closed structure $X \dashv X$ used in CBI \cite{CW}. The special Frobenius algebra provides ``wire bending'' without requiring the full adjunction $\varepsilon \dashv \eta$. Only the Frobenius equations and $\delta;\mu = \id$ are needed.
\end{remark}

\section{Bag-Tabulations}
\label{sec:bag-tab}
%% ============================================================

\begin{definition}[Bag-tabulation]\label{def:bag-tab}
A \emph{bag-tabulation} of an arrow $r \colon X \to Y$ consists of an object $Z$ and arrows $f \colon Z \to X$, $g \colon Z \to Y$ such that $r = f^\op; g$, where $f^\op$ is the Frobenius transpose (Definition~\ref{def:transpose}). In $\Mat_\N$, this means: $Z$ is a finite set, $f$ and $g$ are functions, and $r(x,y) = |\{z \in Z : f(z) = x,\, g(z) = y\}|$.
\end{definition}

\begin{remark}
Unlike CBI's tabulations, we do \emph{not} require the pair $(f,g)$ to be jointly monic. Joint monicity would force $r(x,y) \in \{0,1\}$, recovering the set-semantic (relational) case.
\end{remark}

\subsection{The ordering theorem}

\begin{theorem}[Bag-tabulation ordering]\label{thm:bag-tab-order}
Let $r = f^\op;g$ and $r' = f'^\op;g'$ be bag-tabulations. If there exists an injective map $h \colon Z \hookrightarrow Z'$ with $h;f' = f$ and $h;g' = g$, then $r \leq r'$.
\end{theorem}

\begin{proof}
\begin{align*}
  r &= f^\op;g = (h;f')^\op;(h;g') = f'^\op; h^\op; h; g' \leq f'^\op; \id; g' = r'.
\end{align*}
The third step uses $(r;s)^\op = s^\op;r^\op$. The inequality uses $h^\op = h^*$ (Lemma~\ref{lem:transpose-props}(iv), since $h$ is injective) and $h^*;h \leq \id$ (counit of $h \dashv h^*$).
\end{proof}

\subsection{Functional completeness and existence}

\begin{definition}[Functional completeness]\label{def:func-complete}
A bag-cartesian bicategory $\mathbf{B}$ is \emph{functionally complete} if for every arrow $q \colon X \to I$, there exists an object $Z$ and an arrow $i \colon Z \to X$ such that $q = i^\op;\varepsilon_Z$.
\end{definition}

\begin{theorem}[Existence of bag-tabulations]\label{thm:bag-tab-exist}
In a functionally complete bag-cartesian bicategory, every arrow $r \colon X \to Y$ has a bag-tabulation.
\end{theorem}

\begin{proof}
Form $p = \Phi(r) \colon X \otimes Y \to I$. By functional completeness, there exists $(Z, i \colon Z \to X \otimes Y)$ with $p = i^\op;\varepsilon$. Set $f = i;\pi_X$ and $g = i;\pi_Y$. We show $r = f^\op;g$.

Since $r = \Psi(\Phi(r)) = \Psi(p)$, it suffices to show $\Psi(i^\op;\varepsilon) = f^\op;g$. Write $i = \langle f, g \rangle = \delta_Z;(f \otimes g)$. By contravariance of the transpose: $i^\op = (f^\op \otimes g^\op);\mu_Z$, so $i^\op;\varepsilon = (f^\op \otimes g^\op);\mu;\varepsilon$. Substituting into $\Psi$ and tracing wires: $\eta;\delta$ on $Y$ creates copies $y_1 = y_2$; the copy $y_1$ enters $g^\op$; the spider $\mu;\varepsilon$ contracts the $Z$-outputs of $f^\op$ and $g^\op$ to a single $z$; and $y_2$ passes to the output. The result is $\sum_z [f(z)=x]\cdot [g(z)=y] = (f^\op;g)(x,y)$.
\end{proof}

\section{The Characterisation Theorem}
\label{sec:characterisation}
%% ============================================================

\begin{theorem}[Characterisation]\label{thm:characterisation}
Let $\mathbf{B}$ be a functionally complete bag-cartesian bicategory. Then $\mathbf{B}$ is biequivalent to $\Span_{\mathrm{mono}}(\Map(\mathbf{B}))$.
\end{theorem}

\begin{proof}
Send each arrow $r \colon X \to Y$ to its bag-tabulation span $(Z, f, g)$.

\medskip\noindent\textbf{Essential surjectivity.} Both bicategories have the same objects.

\medskip\noindent\textbf{Full faithfulness ($\Leftarrow$).} Theorem~\ref{thm:bag-tab-order}.

\medskip\noindent\textbf{Full faithfulness ($\Rightarrow$).} From $r \leq r'$, the isomorphism $\Phi$ gives $p \leq p'$, i.e., $|\mathrm{fibre}_i(x,y)| \leq |\mathrm{fibre}_{i'}(x,y)|$ for all $(x,y)$. By the choice principle in the base category, fibrewise injections assemble into a global injection $h \colon Z \hookrightarrow Z'$ with $h;i' = i$.

\medskip\noindent\textbf{Functoriality.} The Cauchy--Schwarz axiom (iii) ensures pullback composition of spans is compatible with the ordering. The witnessing mono for $(f \Had f);(g \Had g) \leq (f;g) \Had (f;g)$ is the tuple reorganisation $((z_1,z_2),(z_3,z_4)) \mapsto ((z_1,z_3),(z_2,z_4))$.
\end{proof}

\section{Completeness and Decidability}
\label{sec:completeness}
%% ============================================================

The characterisation theorem and the algebraic structure of CQ polynomials yield two independent results: a \emph{structural} completeness theorem (the axioms capture exactly the right precongruence) and an \emph{effective} decidability theorem (with explicit complexity bounds). We present both.

\subsection{Completeness of the axiom system}

\begin{corollary}[Completeness]\label{cor:completeness}
The bag-cartesian bicategory axioms \textup{(i)--(v)} are complete for $\Mat_\N$: for CQ terms $\varphi$ and $\psi$, $\varphi \leq_{\mathrm{bag}} \psi$ iff $\varphi \leq \psi$ is derivable from the axioms.
\end{corollary}

\begin{proof}
$\Mat_\N = \Span_{\mathrm{mono}}(\FinSet)$ is functionally complete: given $q \colon X \to I$, take $Z = \{(x,k) : 1 \leq k \leq q(x)\}$ with $i(x,k) = x$. By Theorem~\ref{thm:characterisation}, the axiom-generated precongruence coincides with the $\Span_{\mathrm{mono}}$ ordering, which is bag containment.
\end{proof}

This is a structural result: it identifies the \emph{right axiom system} for bag semantics, placing it in the Carboni--Walters programme alongside the characterisation of $\Rel$ by cartesian bicategories \cite{CW,BSS}. It does not, by itself, yield an effective decision procedure, since arguing that derivability in the symmetric monoidal PROP is recursively enumerable requires care with the infinitely many coherence axiom schemas.

\subsection{Decidability via real closed fields}

An independent route to decidability avoids the axiom system entirely, using instead the algebraic structure of CQ evaluation polynomials.

\begin{theorem}[Decidability]\label{thm:decidability}
Bag-semantic CQ containment is decidable.
\end{theorem}

\begin{proof}
The evaluation polynomial $P_\varphi(T,a) = \sum_{\bar{b}} \prod_j T(v_j, w_j)$ is \textbf{homogeneous} of degree $m = |\mathrm{atoms}(\varphi)|$ in the matrix entries $T(x,y)$.

\medskip\noindent\textbf{Unequal-degree case.}
If $|\mathrm{atoms}(\varphi)| \neq |\mathrm{atoms}(\psi)|$, say $m > m'$, then $P_\varphi(\lambda I, a) = \lambda^m P_\varphi(I,a)$ grows faster than $P_\psi(\lambda I, a) = \lambda^{m'} P_\psi(I,a)$ as $\lambda \to \infty$, so containment fails. The case $m < m'$ is symmetric. So the unequal-degree case is decidable (in constant time).

\medskip\noindent\textbf{Equal-degree case.}
When $m = m'$, the difference $P_\psi - P_\varphi$ is homogeneous of degree $m$. For homogeneous polynomials, $P(\lambda x) = \lambda^m P(x)$, so non-negativity on $\N^k$ is equivalent to non-negativity on $\mathbb{R}_{\geq 0}^k$ (scale rationals to integers; extend to reals by density).

It remains to handle the unbounded domain size $n$. Each variable assignment $h \colon \mathrm{vars}(\varphi) \to [n]$ induces a partition $\pi$ of $\mathrm{vars}(\varphi)$ (identifying variables with the same image). For a partition $\pi$ with $b$ blocks, the contribution involves only the $b^2$ matrix entries between $b$ distinct domain elements, multiplied by $n^{\underline{b}} = n(n{-}1)\cdots(n{-}b{+}1)$ (counting injective placements). Since the number of partitions is finite (bounded by the Bell number of $|\mathrm{vars}(\varphi)| + |\mathrm{vars}(\psi)|$), we obtain:
\begin{enumerate}
  \item For each $n \leq B = |\mathrm{vars}(\varphi)| + |\mathrm{vars}(\psi)|$: a non-negativity condition on $\mathbb{R}_{\geq 0}^{n^2}$.
  \item For $n > B$: an asymptotic condition determined by the leading partition types, again a non-negativity condition on $\mathbb{R}_{\geq 0}^{B^2}$.
\end{enumerate}
Each condition is a first-order sentence $\forall \bar{x} \geq 0\colon P(\bar{x}) \geq 0$ in the theory of \textbf{real closed fields} (RCF). By Tarski's quantifier elimination theorem \cite{Tarski}, each is decidable.
\end{proof}

\begin{remark}[Complexity bounds]
Each RCF sentence involves $O(B^2)$ variables and degree $O(m)$. By the known complexity of quantifier elimination for RCF with a single universal block, the decision procedure runs in space polynomial in $B$ and $m$ for fixed quantifier structure, giving an upper bound of $\mathrm{EXPTIME}$ in the CQ size. A lower bound of $\mathrm{coNP}$ follows from the NP-completeness of set-semantic CQ containment (which reduces to the bag case by restricting to $\{0,1\}$-databases). We conjecture that the precise complexity is $\mathrm{coNP}$-complete.
\end{remark}

\begin{remark}[Two independent results]
Corollary~\ref{cor:completeness} and Theorem~\ref{thm:decidability} are logically independent. The completeness theorem identifies the \emph{right axiom system} for bag semantics (a structural contribution within the Carboni--Walters programme); decidability follows as a consequence, contingent on the effectiveness of derivation search in the symmetric monoidal PROP. The Tarski argument establishes decidability \emph{directly} from the algebraic structure of CQ polynomials, with explicit complexity bounds, but makes no reference to the axiom system. The two results are complementary: the axioms explain \emph{why} bag containment has the structure it does; the Tarski argument shows \emph{how} to decide it.
\end{remark}

\section{The Mono-Stability Principle}
\label{sec:mono-stability}
%% ============================================================

The axioms of Definition~\ref{def:bag-cart} arise systematically: $\Span(\mathcal{E})$ is a cartesian bicategory \cite{CKWW}; each axiom is witnessed by a 2-cell; we retain exactly those whose witness is a monomorphism.

\medskip
\begin{center}
\begin{tabular}{lll}
\hline
\textbf{Axiom} & \textbf{Witness in $\Span$} & \textbf{Mono?} \\
\hline
$\delta \dashv \mu$ (counit) & Diagonal $\Delta \colon X \hookrightarrow X \times X$ & Yes \\
$\varepsilon \dashv \eta$ (unit) & Diagonal $\Delta \colon X \hookrightarrow X \times X$ & Yes \\
$\varepsilon \dashv \eta$ (counit) & $! \colon X \to \{*\}$ & \textbf{No} \\
Lax nat.\ of $\delta$ & $s \mapsto (s,s)$, diagonal embedding & Yes \\
Lax nat.\ of $\varepsilon$ & $f \colon S \to X$ & \textbf{No} \\
Cauchy--Schwarz & Fibrewise pairs $\hookrightarrow$ global pairs & Yes \\
\hline
\end{tabular}
\end{center}

\medskip\noindent The two failures involve non-injective maps: the terminal map $! \colon X \to \{*\}$ and the structure map $f \colon S \to X$ (non-injective when fibres have size $> 1$).

%% ============================================================
\section{The Role of Each Axiom}
\label{sec:roles}
%% ============================================================

\begin{enumerate}[label=\textbf{(\roman*)},leftmargin=2.5em]
  \item \textbf{Frobenius equations.} Wire bending: the Frobenius transpose $r^\op$ and the isomorphism $\Phi/\Psi$ (Theorem~\ref{thm:frob-iso}). Replaces compact closedness.

  \item \textbf{Lax naturality of $\delta$} (bag-lax: $f \leq f \Had f$). Multiplicity structure: $n \leq n^2$ in $\N$. Prevents fractional fibres in bag-tabulations.

  \item \textbf{Cauchy--Schwarz.} Compositional coherence: pullback of bag-tabulations respects the Hadamard product. Ensures functoriality of the characterisation.

  \item \textbf{Special law} ($\delta;\mu = \id$). Makes wire bending involutive. Gives spider normal forms.

  \item \textbf{Unit-of-adjunction} ($\id \leq \varepsilon;\eta$). Slack for functional completeness: tabulating objects can carry multiplicities $> 1$. Semantically: ``forgetting and re-summoning a variable can only increase the count.''
\end{enumerate}

\section{Remarks}
\label{sec:remarks}
%% ============================================================

\textbf{Choice principle.} The converse direction of Theorem~\ref{thm:characterisation} uses fibrewise choice in the base category. This holds in $\FinSet$ and any Boolean topos.

\medskip\noindent\textbf{Functional completeness.} For $\Mat_\N$, this holds by explicit construction. For the free bag-cartesian bicategory on a signature, it should follow from the Kleisli construction for symmetric comonads \cite{CW}.

\medskip\noindent\textbf{Generalisation.} For any class of monomorphisms $\mathcal{M}$ in a category $\mathcal{E}$, one obtains $\Span_\mathcal{M}(\mathcal{E})$ and the $\mathcal{M}$-stable fragment of the cartesian bicategory axioms. The bag-cartesian case is $\mathcal{M} = $ all monos; the set-semantic case is $\mathcal{M} = $ all arrows.
