\title{\textbf{Bag-Cartesian Bicategories:\\ The Mono-Stable Fragment of Span}}
\author{}
\date{}

\begin{document}
\maketitle

\begin{abstract}
We introduce the notion of \emph{bag-cartesian bicategory}: the mono-stable fragment of the cartesian bicategory axioms of Carboni--Walters. The motivating example is $\Mat_\N \cong \Span_{\mathrm{mono}}(\FinSet)$, the bicategory of $\N$-valued matrices modelling bag semantics of conjunctive queries. We develop the theory of \emph{bag-tabulations} (spans of functions, ordered by injective comparison maps), prove a semantic characterisation of the ordering in $\Mat_\N$, and establish structural completeness: the Frobenius equations derive all bag containments that are witnessed by atom-surjective homomorphisms. We conjecture that the full axiom system is complete for $\Mat_\N$, which would resolve the Chaudhuri--Vardi problem.
\end{abstract}

\section{Introduction}
%% ============================================================

The containment problem for conjunctive queries (CQs) under bag semantics has been open since Chaudhuri and Vardi raised it in 1993 \cite{CV}. Under set semantics, the problem is decidable and elegantly characterised: containment coincides with the existence of a homomorphism between query hypergraphs (Chandra--Merlin \cite{CM}), which in turn coincides with derivability in the equational theory of cartesian bicategories (Bonchi--Seeber--Soboci\'nski \cite{BSS}). Under bag semantics, no such characterisation has been known.

We develop a framework that extends the cartesian bicategory approach to bag semantics. The central observation is that $\Mat_\N$, the bicategory of $\N$-valued matrices modelling bag semantics, arises as $\Span_{\mathrm{mono}}(\FinSet)$: the locally posetal bicategory obtained from $\Span(\FinSet)$ by restricting 2-cells to monomorphisms. The axioms of what we call a \emph{bag-cartesian bicategory} are precisely the \emph{mono-stable fragment} of the cartesian bicategory axioms of Carboni--Walters \cite{CW,CKWW}: those axioms whose witnessing 2-cells in $\Span$ are necessarily injective.

Our main results are: (1) a semantic characterisation of the ordering in $\Mat_\N$ via bag-tabulations (Theorem~\ref{thm:characterisation}); (2) structural completeness: the Frobenius equations derive all bag containments witnessed by atom-surjective homomorphisms (Theorem~\ref{thm:structural-complete}); and (3) a completeness conjecture (Conjecture~\ref{conj:completeness}) asserting that the full axiom system captures all bag containments, which would imply decidability of the Chaudhuri--Vardi problem.

\subsection{Conventions}

Throughout, $\mathbf{B}$ denotes a locally posetal bicategory with symmetric monoidal structure $(\otimes, I)$. We write composition in \textbf{diagrammatic order}: $f;g$ means ``first $f$, then $g$.'' We write $f \colon A \to B$ for an arrow with source $A$ and target $B$.

An arrow $f \colon A \to B$ is a \emph{map} if it has a right adjoint $f^* \colon B \to A$ satisfying
\[
  \text{unit: } \id_A \leq f;f^*, \qquad
  \text{counit: } f^*;f \leq \id_B.
\]
We write $\Map(\mathbf{B})$ for the locally full sub-bicategory of maps.

\section{Bag-Cartesian Bicategories}
\label{sec:def}
%% ============================================================

We build the definition in stages: first the Frobenius structure, then the ordering axioms.

\subsection{Special commutative Frobenius structure}

\begin{definition}[Frobenius structure]\label{def:frob}
A \emph{commutative special Frobenius structure} on an object $X$ of $\mathbf{B}$ consists of four arrows
\[
  \delta_X \colon X \to X \otimes X, \quad
  \mu_X \colon X \otimes X \to X, \quad
  \varepsilon_X \colon X \to I, \quad
  \eta_X \colon I \to X
\]
satisfying the following equations.

\begin{enumerate}[label=(\alph*),leftmargin=2em]
  \item \textbf{Comonoid.} $(\delta_X, \varepsilon_X)$ is a cocommutative comonoid:
  \begin{gather*}
    \delta;(\delta \otimes \id) = \delta;(\id \otimes \delta), \qquad
    \delta;\sigma = \delta, \\
    \delta;(\varepsilon \otimes \id) = \id = \delta;(\id \otimes \varepsilon),
  \end{gather*}
  where $\sigma \colon X \otimes X \to X \otimes X$ is the symmetry.

  \item \textbf{Monoid.} $(\mu_X, \eta_X)$ is a commutative monoid:
  \begin{gather*}
    (\mu \otimes \id);\mu = (\id \otimes \mu);\mu, \qquad
    \sigma;\mu = \mu, \\
    (\eta \otimes \id);\mu = \id = (\id \otimes \eta);\mu.
  \end{gather*}

  \item \textbf{Frobenius law.}
  \[
    (\delta \otimes \id);(\id \otimes \mu) = \mu;\delta = (\id \otimes \delta);(\mu \otimes \id).
  \]

  \item \textbf{Special law.} $\delta;\mu = \id_X$.
\end{enumerate}
\end{definition}

\begin{remark}\label{rmk:spider}
In a commutative special Frobenius algebra, every connected string diagram with $m$ inputs and $n$ outputs is equal to a unique \emph{spider} $\mathrm{Spider}(m,n)$. This is the ``spider theorem,'' which gives a normal form for connected diagrams. In particular, $\mu;\varepsilon = \mathrm{Spider}(2,0)$ and $\eta;\delta = \mathrm{Spider}(0,2)$.
\end{remark}

\subsection{Deterministic arrows and the Frobenius transpose}

\begin{definition}[Deterministic arrow]\label{def:det}
An arrow $f \colon X \to Y$ is \emph{deterministic} if it is a comonoid homomorphism:
\[
  f;\delta_Y = \delta_X;(f \otimes f), \qquad f;\varepsilon_Y = \varepsilon_X.
\]
We write $\Det(\mathbf{B})$ for the subcategory of deterministic arrows. In $\Mat_\N$, the deterministic arrows are precisely the functions (i.e., $\{0,1\}$-matrices with exactly one $1$ per row).
\end{definition}

\begin{remark}\label{rmk:det-vs-map}
In $\Mat_\N$, every map is deterministic (verified by direct computation). The converse fails: a non-injective function $f$ is deterministic but is \emph{not} a map, since the counit $f^*;f \leq \id$ requires $|f^{-1}(b)| \leq 1$.
\end{remark}

\begin{definition}[Frobenius transpose]\label{def:transpose}
For an arrow $r \colon X \to Y$, the \emph{Frobenius transpose} $r^\op \colon Y \to X$ is defined by
\[
  r^\op \;=\; ((\eta_X;\delta_X) \otimes \id_Y);(\id_X \otimes r \otimes \id_Y);(\id_X \otimes (\mu_Y;\varepsilon_Y))
\]
where $\eta_X;\delta_X \colon I \to X \otimes X$ creates a ``cup'' and $\mu_Y;\varepsilon_Y \colon Y \otimes Y \to I$ creates a ``cap.'' Equivalently (by the spider theorem), $r^\op$ is obtained by rotating the string diagram of $r$ by $180^\circ$.
\end{definition}

\begin{lemma}\label{lem:transpose-props}
The Frobenius transpose satisfies:
\begin{enumerate}[label=(\roman*)]
  \item $(r^\op)^\op = r$ \quad \textup{(involution)},
  \item $(r;s)^\op = s^\op; r^\op$ \quad \textup{(contravariance)},
  \item $\id^\op = \id$,
  \item for an injective map $h$: $h^\op = h^*$ \quad \textup{(transpose $=$ right adjoint)}.
\end{enumerate}
\end{lemma}

\begin{proof}
Properties (i)--(iii) follow from the spider theorem. For~(iv): in $\Mat_\N$, $h^\op = h^T$ (matrix transpose), and one verifies $(h;h^T)(a,a') = [h(a){=}h(a')] \geq [a{=}a']$ and $(h^T;h)(b,b') = |h^{-1}(b)| \cdot [b{=}b'] \leq [b{=}b']$ (using injectivity). Since right adjoints in a locally posetal bicategory are unique, $h^\op = h^*$.
\end{proof}

\subsection{The Hadamard product}

\begin{definition}[Hadamard product]\label{def:hadamard}
For arrows $f, g \colon X \to Y$, the \emph{Hadamard product} is
\[
  f \Had g \;=\; \delta_X;(f \otimes g);\mu_Y \;\colon\; X \to Y.
\]
In $\Mat_\N$, $(f \Had g)(x,y) = f(x,y) \cdot g(x,y)$ \textup{(entrywise multiplication)}.
\end{definition}

\subsection{The axioms}

\begin{definition}[Bag-cartesian bicategory]\label{def:bag-cart}
A locally posetal bicategory $\mathbf{B}$ with symmetric monoidal structure $(\otimes, I)$ is \emph{bag-cartesian} if:
\begin{enumerate}[label=\textbf{(\roman*)},leftmargin=2.5em]
  \item \textbf{Frobenius structure.} Each object $X \in \ob(\mathbf{B})$ carries a commutative special Frobenius structure $(\delta_X, \mu_X, \varepsilon_X, \eta_X)$ as in Definition~\ref{def:frob}.

  \item \textbf{Lax naturality of $\delta$.} For every arrow $r \colon X \to Y$:
  \[
    r;\delta_Y \;\leq\; \delta_X;(r \otimes r).
  \]

  \item \textbf{Cauchy--Schwarz.} For all $f \colon X \to Y$ and $g \colon Y \to Z$:
  \[
    (f \Had f);(g \Had g) \;\leq\; (f;g) \Had (f;g).
  \]

  \item \textbf{Adjunction $\delta \dashv \mu$.} For all $X$:
  \[
    \delta_X;\mu_X = \id_X \quad \textup{(unit/special law)}, \qquad
    \mu_X;\delta_X \leq \id_{X \otimes X} \quad \textup{(counit)}.
  \]

  \item \textbf{Unit-of-adjunction.} For all $X$:
  \[
    \id_X \;\leq\; \varepsilon_X;\eta_X.
  \]
\end{enumerate}
\end{definition}

\begin{remark}\label{rmk:bag-lax}
Composing axiom (ii) on the right with $\mu$ and applying $\delta;\mu = \id$ yields the \emph{bag-lax} inequality $r \leq r \Had r$. The converse does not hold in general.
\end{remark}

\begin{remark}[What is \emph{not} assumed]
We do \textbf{not} assume the full adjunction $\varepsilon \dashv \eta$ (only the unit half, axiom (v)); the counit $\eta;\varepsilon \leq \id_I$ fails in $\Mat_\N$ for $|X| > 1$. We do not assume lax naturality of $\varepsilon$. Note that discreteness in the sense of \cite{CW} ($\mu;\delta = (\delta \otimes \id);(\id \otimes \mu)$) \emph{does} hold: it is the Frobenius law.
\end{remark}

\subsection{The motivating example}

\begin{example}[$\Mat_\N$]\label{ex:mat-n}
The bicategory $\Mat_\N$ has finite sets as objects, $\N$-valued matrices as arrows (composed by matrix multiplication), $X \otimes Y = X \times Y$, and entrywise ordering. The Frobenius structure is: $\delta(a,(b,c)) = [a{=}b{=}c]$, $\mu((a,b),c) = [a{=}b{=}c]$, $\varepsilon(a,*) = 1$, $\eta(*,a) = 1$. One verifies axioms (i)--(v). The deterministic arrows are functions ($\Det(\Mat_\N) = \FinSet$); the maps are injective functions ($\Map(\Mat_\N) = \FinSet_{\mathrm{mono}}$).
\end{example}

\begin{proposition}\label{prop:span-mono}
$\Mat_\N \cong \Span_{\mathrm{mono}}(\FinSet)$ as locally posetal bicategories.
\end{proposition}

\begin{proof}
A span $X \xleftarrow{f} S \xrightarrow{g} Y$ determines the matrix $M(x,y) = |\{s : f(s) = x,\, g(s) = y\}|$. A mono of spans $\alpha \colon S \hookrightarrow S'$ exists iff $M \leq M'$ entrywise. Composition via pullback gives matrix multiplication.
\end{proof}

\section{The Frobenius Isomorphism}
\label{sec:frob-iso}
%% ============================================================

\begin{theorem}[Frobenius Isomorphism]\label{thm:frob-iso}
The map $\Phi(s) = (s \otimes \id);\mu;\varepsilon \colon \mathbf{B}(X,Y) \to \mathbf{B}(X \otimes Y, I)$ is an order-isomorphism, with inverse $\Psi(q) = (\id \otimes \eta);(\id \otimes \delta);(q \otimes \id)$.
\end{theorem}

\begin{proof}
Both are order-preserving by monotonicity of $;$ and $\otimes$. For $\Psi(\Phi(s)) = s$: by the spider theorem, $\eta;\delta = \mathrm{Spider}(0,2)$ and $\mu;\varepsilon = \mathrm{Spider}(2,0)$; connected on one leg they give $\mathrm{Spider}(1,1) = \id$. The identity $\Phi(\Psi(q)) = q$ follows by a similar calculation.
\end{proof}

\section{Bag-Tabulations}
\label{sec:bag-tab}
%% ============================================================

\begin{definition}[Bag-tabulation]\label{def:bag-tab}
A \emph{bag-tabulation} of $r \colon X \to Y$ is a triple $(Z, f \colon Z \to X, g \colon Z \to Y)$ of deterministic arrows with $r = f^\op;g$. In $\Mat_\N$: $r(x,y) = |\{z : f(z) = x,\, g(z) = y\}|$.
\end{definition}

\begin{theorem}[Bag-tabulation ordering]\label{thm:bag-tab-order}
If $r = f^\op;g$, $r' = f'^\op;g'$, and there exists an injective map $h \colon Z \hookrightarrow Z'$ with $h;f' = f$ and $h;g' = g$, then $r \leq r'$.
\end{theorem}

\begin{proof}
$r = f^\op;g = (h;f')^\op;(h;g') = f'^\op;h^\op;h;g' \leq f'^\op;g' = r'$, using Lemma~\ref{lem:transpose-props}(ii,iv) and $h^*;h \leq \id$.
\end{proof}

\section{Semantic Characterisation of $\Mat_\N$}
\label{sec:characterisation}
%% ============================================================

\begin{theorem}[Semantic characterisation]\label{thm:characterisation}
In $\Mat_\N$: every arrow has a bag-tabulation, and $r \leq r'$ iff there exists an injective function $h$ between their bag-tabulating sets with $h;f' = f$, $h;g' = g$.
\end{theorem}

\begin{proof}
\textbf{Existence.} Given $r \colon X \to Y$, take $Z = \{(x,y,k) : 1 \leq k \leq r(x,y)\}$ with $f(x,y,k) = x$, $g(x,y,k) = y$. Then $(f^\op;g)(x,y) = |\{(x',y',k) : x' = x, y' = y\}| = r(x,y)$.

\medskip\noindent\textbf{($\Leftarrow$)} Theorem~\ref{thm:bag-tab-order}.

\medskip\noindent\textbf{($\Rightarrow$)} If $r \leq r'$ entrywise, then $|\mathrm{fibre}(x,y)| \leq |\mathrm{fibre}'(x,y)|$ for all $(x,y)$. Fibrewise injections (which exist for finite sets) assemble into a global injection $h \colon Z \hookrightarrow Z'$.
\end{proof}

\section{Structural Completeness}
\label{sec:structural}
%% ============================================================

\begin{theorem}[Structural completeness]\label{thm:structural-complete}
If $\varphi \leq_{\mathrm{bag}} \psi$ is witnessed by an atom-surjective homomorphism $h \colon H_\psi \to H_\varphi$, then $\varphi \leq \psi$ is derivable from the Frobenius equations \textup{(axiom (i))} alone.
\end{theorem}

\begin{proof}[Proof sketch]
An atom-surjective homomorphism $h$ is a map between variable sets that identifies some variables (surjective on both variables and atoms). In the PROP, variable identification corresponds to spider merging: replacing two spiders (variables) connected by $h$ with a single spider, using $\mu;\delta$ (the Frobenius map). Since $h$ fixes the free variable and covers all atoms, the resulting diagram equals $\psi$. The derivation uses only the Frobenius equations and the special law.
\end{proof}

\begin{remark}
In our enumeration of CQs with $\leq 3$ atoms and $\leq 3$ bound variables over one binary relation, $219$ out of $224$ bag containments are witnessed by atom-surjective homomorphisms. These are all derivable by Theorem~\ref{thm:structural-complete}. The remaining $5$ ``gap pairs'' are bag containments without such witnesses.
\end{remark}

\section{The Completeness Conjecture}
\label{sec:conjecture}
%% ============================================================

\begin{conjecture}[Completeness]\label{conj:completeness}
The bag-cartesian bicategory axioms \textup{(i)--(v)} are complete for $\Mat_\N$: for CQ terms $\varphi$ and $\psi$, $\varphi \leq_{\mathrm{bag}} \psi$ iff $\varphi \leq \psi$ is derivable from the axioms.
\end{conjecture}

\begin{remark}[The gap]
The semantic characterisation (Theorem~\ref{thm:characterisation}) shows that bag containment in $\Mat_\N$ is equivalent to the existence of injective comparison maps between bag-tabulations. For \emph{specific} databases, the bag-tabulation ordering theorem (Theorem~\ref{thm:bag-tab-order}) gives a derivation. However, these derivations are \emph{database-specific}: the tabulating spans and comparison injections depend on the matrix entries of the database. A proof of the conjecture would require showing that, for each valid $\varphi \leq_{\mathrm{bag}} \psi$, there exists a \emph{single} derivation in the PROP that works universally for all databases. This is the fundamental difficulty: bag containment witnesses are database-dependent, unlike the structural (database-independent) homomorphisms of set semantics.
\end{remark}

\begin{remark}[Decidability]
If Conjecture~\ref{conj:completeness} holds, bag-semantic CQ containment is decidable: containment is r.e.\ (enumerate derivations in the PROP, where coherence axioms are built into the PROP structure and equality is decidable by the spider theorem), and non-containment is r.e.\ (enumerate counterexample databases).
\end{remark}

\begin{remark}[Evidence]
Computational enumeration supports the conjecture: all $224$ bag containments found for CQs with $\leq 3$ atoms are consistent with the axioms ($219$ derived structurally, $5$ verified semantically). The $5$ gap pairs all follow the ``self-loop to pendant'' pattern, suggesting a uniform derivation strategy via axiom~(v) ($\id \leq \varepsilon;\eta$), though we have not found explicit PROP derivations.
\end{remark}

\section{The Mono-Stability Principle}
\label{sec:mono-stability}
%% ============================================================

The axioms of Definition~\ref{def:bag-cart} arise systematically: $\Span(\mathcal{E})$ is a cartesian bicategory \cite{CKWW}; each axiom is witnessed by a 2-cell; we retain exactly those whose witness is a monomorphism.

\medskip
\begin{center}
\begin{tabular}{lll}
\hline
\textbf{Axiom} & \textbf{Witness in $\Span$} & \textbf{Mono?} \\
\hline
$\delta \dashv \mu$ (unit) & Isomorphism (special law) & Yes \\
$\delta \dashv \mu$ (counit) & Diagonal $\Delta \colon X \hookrightarrow X \times X$ & Yes \\
$\varepsilon \dashv \eta$ (unit) & Diagonal $\Delta \colon X \hookrightarrow X \times X$ & Yes \\
$\varepsilon \dashv \eta$ (counit) & $! \colon X \to \{*\}$ & \textbf{No} \\
Lax nat.\ of $\delta$ & $s \mapsto (s,s)$, diagonal embedding & Yes \\
Lax nat.\ of $\varepsilon$ & $f \colon S \to X$ & \textbf{No} \\
Cauchy--Schwarz & Fibrewise pairs $\hookrightarrow$ global pairs & Yes \\
\hline
\end{tabular}
\end{center}

\medskip\noindent The two failures involve non-injective maps: the terminal map $! \colon X \to \{*\}$ and the structure map $f \colon S \to X$ (non-injective when fibres have size $> 1$).

\section{Remarks}
\label{sec:remarks}
%% ============================================================

\textbf{Relation to CBI.} In the cartesian bicategory framework of \cite{CW}, set-semantic CQ containment $\varphi \subseteq_{\mathrm{set}} \psi$ is equivalent to the existence of a homomorphism $H_\psi \to H_\varphi$ (Chandra--Merlin \cite{CM}), which is a \emph{structural} (database-independent) witness. The derivation in the cartesian bicategory uses this witness directly \cite{BSS}. Our framework provides the analogous \emph{semantic} characterisation for bags (Theorem~\ref{thm:characterisation}), but the passage from semantics to syntax (the completeness conjecture) remains open because bag containment lacks structural witnesses in general.

\medskip\noindent\textbf{The role of each axiom.} (i)~Frobenius: wire bending and the isomorphism $\Phi/\Psi$. (ii)~Lax naturality of $\delta$: multiplicity structure ($n \leq n^2$ in $\N$). (iii)~Cauchy--Schwarz: compositional coherence of bag-tabulations. (iv)~Adjunction $\delta \dashv \mu$: the counit $\mu;\delta \leq \id$ provides $h^\op;h \leq \id$ for injective maps. (v)~Unit-of-adjunction $\id \leq \varepsilon;\eta$: ``forgetting and re-summoning a variable can only increase the count.''

\medskip\noindent\textbf{Generalisation.} For any class of monomorphisms $\mathcal{M}$ in a category $\mathcal{E}$, one obtains $\Span_\mathcal{M}(\mathcal{E})$ and the $\mathcal{M}$-stable fragment of the cartesian bicategory axioms. The bag-cartesian case is $\mathcal{M} = $ all monos; the set-semantic case is $\mathcal{M} = $ all arrows.
